	\begin{UseCase}{CU21}{Pedir libro a biblioteca}{
		El Bibliotecario podrá pedir prestado un libro a otra biblioteca, esto con la finalidad de satisfacer la demanda de libros por parte de los usuarios. Lo hará seleccionando un tipo de búsqueda e introduciendo uno de los siguientes datos: libro, autor, numero de paginas, fecha de
publicación, editorial o la biblioteca a la cual le pedirá el préstamo y seleccionara el libro que pedirá prestado.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Bibliotecario}
		\UCitem{Propósito}{Pedir un libro a otra biblioteca, mediante los datos del formulario.}
		\UCitem{Entradas}{Seleccionar una de las siguientes opciones:
			\begin{itemize}
				\item Titulo
				\item Autor
				\item Numero de paginas
				\item Fecha de publicación
				\item Editorial
				\item Biblioteca
				
			\end{itemize}
		}
		\UCitem{Salidas}{Una tabla con todos los libros que se encontró con la búsqueda y las bibliotecas que tiene esos libros disponibles así como cuantos ejemplares.Mensaje \MSGref{MSG01}{Operación realizada exitosa mente}.Mensaje\MSGref{E3}{Datos introducidos incorrectamente.}Mensaje\MSGref{MSG03}{La biblioteca no tiene convenio.}}
		\UCitem{Precondiciones}{ 
			\begin{itemize}
				\item Se debe tener convenio con la biblioteca
			\end{itemize}
		}
		\UCitem{Postcondiciones}{Se le mandará un mensaje a la biblioteca donde se especifique el libro que deseamos pedir prestado}
		\UCitem{Autor}{Rodriguez Cervantes Arturo y Monserrat Ceron Rodriguez .}
		\UCitem{Revisó}{Diego Manriquez Molina y Abiran Natanael Salas Hernandez}
	\end{UseCase}
		%-------------------------------------- COMIENZA descripción Trayectoria Principal
	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Del \UCref{CU08} presiona el botón:  \IUbutton{Pedir libro a biblioteca} de la pantalla \IUref{UI08.2}{Pantalla de prestamo interbibliotecario}.
		\UCpaso[\UCsist]Muestra la pantalla \IUref{UI21.1}{pantalla de pedir libro a biblioteca búsqueda}.
		\UCpaso[\UCactor]Selecciona un método de búsqueda. 
		\UCpaso[\UCactor]Ingresa el dato que solicita.
		\UCpaso[\UCactor]Presiona el botón: \IUbutton{Buscar} se verifican reglas de negocio \BRref{RN46}{Convenio con biblioteca} \BRref{RN42}{Datos completos}\Trayref{A} \Trayref{B} \Trayref{C}.
		\UCpaso[\UCsist] Muestra la pantalla \IUref{UI21.2}{pantalla resultados}
		\UCpaso[\UCactor]selecciona que ejemplar desea solicitar, la cantidad de ejemplares que necesita y presiona el botón \IUbutton{Solicitar préstamo}
		\UCpaso[\UCsist] Da el mensaje \MSGref{MSG01}{Operación realizada exitosa mente}
		\UCpaso[\UCactor] presiona el botón \IUbutton{OK}
	\end{UCtrayectoria}
