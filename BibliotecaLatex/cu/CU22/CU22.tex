	\begin{UseCase}{CU22}{Prestar libro a biblioteca}{
		El administrador podrá prestar un libro a otra biblioteca, esto con la finalidad de satisfacer la demanda de libros de otras bibliotecas prestando nuestros libros. Lo hará aceptando las solicitudes de préstamo de libro
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Administrador}
		\UCitem{Propósito}{Prestar un libro a otra biblioteca, Se verificara la solicitud que se le hizo al sistema para prestar el libro y se aceptara o se rechazara la solicitud  se verificara si el libro se puede prestar, y si se tiene convenio con la biblioteca.}
		\UCitem{Entradas}{la petición de préstamo con todos los datos del libro}
		\UCitem{Salidas}{Registro en la base de datos junto con los mensajes .Mensaje \MSGref{MSG01}{Operación realizada exitosa mente}.Mensaje\MSGref{MSG03}{El libro no está disponible.}}
		\UCitem{Precondiciones}{ 
			\begin{itemize}
				\item El libro debe estar disponible para préstamo. 
				\item conexión a la base de datos.
				\item Se debe tener convenio con la biblioteca
			\end{itemize}
		}
		\UCitem{Postcondiciones}{El libro estará en la base de datos como prestado}
		\UCitem{Autor}{Rodriguez Ceravates Arturo y Monserrat Ceron Rodriguez .}
		\UCitem{Revisó}{Diego Manriquez Molina y Abiran Natanael Salas Hernandez}
	\end{UseCase}
