	\begin{UseCase}{CU08}{Préstamo interbibliotecario}{
		Muestra al Bibliotecario la opción de préstamo interbibliotecario y este desplegara las opciones de Pedir libro a biblioteca , Prestar libro a biblioteca y Notificación de préstamo interbibliotecario, al seleccionar alguna mandará a su respectiva pantalla
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Bibliotecario}
		\UCitem{Propósito}{Dar a elegir la opción que el administrador desea hacer, solicitar un libro, responder a peticiones de libros o revisar el estado de las solicitudes de libros que a enviado.}
		\UCitem{Entradas}{Presionar botón préstamo interbibliotecario}
		\UCitem{Salidas}{Patalla correspondiente al botón presionado }
		\UCitem{Precondiciones}{ 
			\begin{itemize}
				\item Ninguna
			\end{itemize}
		}
		\UCitem{Postcondiciones}{El sistema mostrara las pantallas que corresponden dependiendo del botón}
		\UCitem{Autor}{Rodriguez Cervantes Arturo y Monserrat Ceron Rodriguez .}
		\UCitem{Revisó}{Diego Manriquez Molina y Abiran Natanael Salas Hernandez}
	\end{UseCase}
		%-------------------------------------- COMIENZA descripción Trayectoria Principal
	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Presiona el botón:  \IUbutton{Préstamo interbibliotecario} de la pantalla \IUref{UI08.1}{Pantalla de inicio}.
		\UCpaso[\UCsist] Muestra la pantalla \IUref{UI08.2}{pantalla de préstamo interbibliotecario}.
		\UCpaso[\UCactor] Selecciona  una opción de préstamo interbibliotecario dando clic en alguno de los botones: \IUbutton{Pedir libro a biblioteca} \IUbutton{Préstamo de libro a Biblioteca} \IUbutton{Notificación de préstamo interbibliotecario}
		\UCpaso[\UCsist] Muestra la pantalla correspondiente de cada botón: Pedir libro a biblioteca \UCref{CU21} , Prestar libro a biblioteca \UCref{CU22} y Notificación de préstamo interbibliotecario \UCref{CU23}
	\end{UCtrayectoria}
