
	\begin{UseCase}{CU06}{Asociar un Estado físico a un Libro.}{
		Los libros por el uso pueden presentar un daño físico  impidiendo que sea prestado, el registrar las condiciones de un libro corresponde al bibliotecario que evalúa si es necesario hacer el registro del estado de un determinado libro. 
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Bibliotecario}
		\UCitem{Propósito}{Asociar un estado Físico a un libro en particular para considerar si está en las condiciones necesarias para su préstamo o mantenimiento. }
		\UCitem{Entradas}{Identificador del libro, Fecha en la que se asocia el estado al libro y el estado que se le asocia al libro. }
		\UCitem{Salidas}{Registro con una fecha asociada de cuando se notificó el estado del libro.  }
		\UCitem{Precondiciones}{ 
			\begin{itemize}
			\item El libro a asociar el estado debe tener registro en el sistema. 
			\end{itemize}
		}
		\UCitem{Postcondiciones}{Los estados asociados del libro se registran para considerar en los préstamos futuros o reparaciones. }
		\UCitem{Autor}{
			\begin{itemize}
				\item Bautista Rosales Mauricio
				\item Salgado Carlos
			\end{itemize}
			}
		\UCitem{Revisión}{
			\begin{itemize}
				\item Cortés Pérez Edy 
				\item Marín Gárcia Fernando
			\end{itemize}
			}
		\UCitem{Estado}{En Revisión}
		
	\end{UseCase}
	\begin{UCtrayectoria}{Principal}

	\UCpaso[\UCactor] El bibliotecario solicita registrar un estado a un libro  presionando el botón  \IUbutton{Asociar Estado Físico} de la interfaz principal.      
	\UCpaso[\UCsist]  consulta los estados que pueden asociarse a un libro.  
	\UCpaso[\UCsist]  muestra la interfaz \IUref{EstadoFisico}.
	\UCpaso[\UCactor] El bibliotecario ingresa el identificador del libro. 
	\UCpaso[\UCactor] El bibliotecario selecciona el estado que desea asociar al libro.
	\UCpaso[\UCactor] El bibliotecario ingresa observaciones asociadas al estado. 
	\UCpaso[\UCactor] El bibliotecario comienza el registro presionando el botón \IUbutton{Registrar} 
	\UCpaso[\UCsist]  El sistema valida la existencia del libro regla del \BRref{RN31}. \Trayref{A}
	\UCpaso[\UCactor] El administrador confirma los datos presionando el botón  \IUbutton{OK}
	\UCpaso[\UCsist]  registra el cambio de estado asociando la fecha actual. 
	\UCpaso[\UCsist]  muestra el Mensaje  \MSGref{MSG1}
	\UCpaso[\UCactor] El bibliotecario confirma presionando el botón \IUbutton{OK}

			
	\end{UCtrayectoria}					
		\begin{UCtrayectoriaA}{A}{No existe el registro del libro (el identificador es incorrecto) }
			\UCpaso[\UCsist] El sistema muestra el mensaje   \MSGref{E4}
			\UCpaso[\UCactor] El bibliotecario confirma presionando el botón \IUbutton{Aceptar}
	\end{UCtrayectoriaA}
