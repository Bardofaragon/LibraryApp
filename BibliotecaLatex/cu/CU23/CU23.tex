	\begin{UseCase}{CU23}{Notificación de préstamo interbibliotecario}{
		El administrador podrá consultar todas la solicitudes de prestamo a otras bibliotecas que ha echo para saber cuales ya han sido autorizadas y cuales no esto lo hace mostrando una tabla y el estatus en que se encuentra la peticion Pendiente o Autorizada
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Administrador}
		\UCitem{Propósito}{Visualizar las solicitudes de libro que han sido autorizadas por la otra biblioteca.}
		\UCitem{Entradas}{Ninguna}
		\UCitem{Salidas}{Titulo, autor, numero de paginas, fecha de publicacion editorial, biblioteca, estatus.Mensaje \MSGref{MSG01}{Operación realizada exitosa mente}.Mensaje\MSGref{MSG03}{El libro no está disponible.}}
		\UCitem{Precondiciones}{ 
			\begin{itemize}
				\item El administrador debe estar logueado en el sistema. 
				\item conexión a la base de datos.
			\end{itemize}
		}
		\UCitem{Postcondiciones}{Ninguna}
